# Knapsack vs Container Loading

weights = [10, 20, 30]
values = [60, 100, 120]
capacity = 50

# 0/1 Knapsack using Dynamic Programming
n = len(weights)
dp = [[0 for _ in range(capacity + 1)] for _ in range(n + 1)]

for i in range(1, n + 1):
    for w in range(capacity + 1):
        if weights[i - 1] <= w:
            dp[i][w] = max(
                values[i - 1] + dp[i - 1][w - weights[i - 1]],
                dp[i - 1][w]
            )
        else:
            dp[i][w] = dp[i - 1][w]

print("Knapsack Maximum Value =", dp[n][capacity])

# Fractional Container Loading using Greedy
items = []

for i in range(n):
    ratio = values[i] / weights[i]
    items.append((ratio, weights[i], values[i]))

items.sort(reverse=True)

total_value = 0
remaining = capacity

for ratio, weight, value in items:
    if remaining >= weight:
        total_value += value
        remaining -= weight
    else:
        total_value += ratio * remaining
        break

print("Container Loading Value =", total_value)
